CREATE TABLE users(
id SERIAL,
email varchar(255),
password varchar(255),
PRIMARY KEY (id));

CREATE TABLE homeProperties(
id SERIAL,
owner INT REFERENCES users(id) ON DELETE CASCADE,
name varchar(255),
country varchar(30),
PRIMARY KEY (id));

CREATE TABLE others(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
hjoht DOUBLE PRECISION,
cost DOUBLE PRECISION,
pipe BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE groundFloor(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
uValue  DOUBLE PRECISION,
area DOUBLE PRECISION,
materials varchar(255),
protected BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE outerWall(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
uValue DOUBLE PRECISION,
area DOUBLE PRECISION,
materials varchar(255),
protected BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE roofConstruction(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
uValue DOUBLE PRECISION,
area DOUBLE PRECISION,
materials varchar(255),
protected BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE windowAll(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
PRIMARY KEY (id));

CREATE TABLE doors(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
PRIMARY KEY (id));

CREATE TABLE windowSingle(
id SERIAL,
windowAll INT REFERENCES windowAll(id) ON DELETE CASCADE,
uValue DOUBLE PRECISION,
area DOUBLE PRECISION,
materials varchar(255),
bridgeValue DOUBLE PRECISION,
name varchar(255),
protected BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE door(
id SERIAL,
doors INT REFERENCES doors(id) ON DELETE CASCADE,
uValue DOUBLE PRECISION,
area DOUBLE PRECISION,
materials varchar(255),
bridgeValue DOUBLE PRECISION,
name varchar(255),
protected BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE bridges(
id SERIAL,
properties INT REFERENCES homeProperties(id) ON DELETE CASCADE,
outerWallToOuterWall DOUBLE PRECISION,
outerWallToRoof DOUBLE PRECISION,
outerWallToMiddleBasement DOUBLE PRECISION,
outerWallToGroundFloor DOUBLE PRECISION,
protected BOOLEAN,
cond INT,
PRIMARY KEY (id));

CREATE TABLE defaults(
id SERIAL,
part varchar(255),
materials varchar(255),
country varchar(30),
era INT,
values DOUBLE PRECISION,
PRIMARY KEY (id));
